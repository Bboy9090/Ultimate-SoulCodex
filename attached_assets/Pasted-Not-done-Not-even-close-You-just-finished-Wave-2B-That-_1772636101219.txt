Wave 3 Interpretation Engine

src/codex/interpret.ts

export function buildCodexReading(chart:any): CodexReading {

  const sun = chart.planets.sun.sign
  const moon = chart.planets.moon.sign
  const rising = chart.houses?.ascSign ?? "Unknown"

  const identity = `My core identity expresses through ${sun} energy. I naturally focus on improvement, refinement, and meaningful contribution.`

  const emotionalNature = `My emotional world is shaped by the Moon in ${moon}. I process feelings internally and often reflect deeply before reacting.`

  const decisionStyle = `My instinct in complex situations is to observe, analyze, and act with intention rather than impulse.`

  return {
    identity,
    emotionalNature,
    decisionStyle,
    strengths: [
      "Pattern recognition",
      "Strategic thinking",
      "Loyalty to meaningful work"
    ],
    growthEdges: [
      "Overthinking",
      "Taking responsibility for too much",
      "Needing solitude to recharge"
    ],
    archetype: "Quiet Stone Elder",
    archetypeDescription:
      "A stabilizing presence who builds foundations others rely on."
  }
}

    ok:true,
    reading
  })
}
