export type PlanetKey =
  | "sun" | "moon" | "mercury" | "venus" | "mars"
  | "jupiter" | "saturn" | "uranus" | "neptune" | "pluto"
  | "north_node";

export type AspectKey = "conjunction" | "opposition" | "square" | "trine" | "sextile";

export interface PlanetPosition {
  key: PlanetKey;
  lon: number; // 0..360 ecliptic longitude
  sign: string; // Aries..Pisces
  degInSign: number; // 0..29.999
  house?: number; // 1..12 if time known + houses present
}

export interface HouseData {
  system: "placidus" | "whole_sign";
  cusps: number[]; // length 12, degrees 0..360 (cusp 1..12)
  asc?: number; // ascendant longitude if provided

}

export interface AspectHit {
  a: PlanetKey;
  b: PlanetKey;
  aspect: AspectKey;
  orb: number; // degrees away from exact
  exactDeg: number; // 0, 60, 90, 120, 180
}

export interface FullChart {
  meta: {
    dateISO: string;
    time24?: string;
    timeUnknown: boolean;
    place: string;
    lat: number;
    lon: number;
    timezone?: string;
    utcOffsetMin?: number;
    isDST?: boolean;
  };
  planets: Record<PlanetKey, PlanetPosition>;
  houses?: HouseData;
  aspects: AspectHit[];
}

5) Zodiac helpers

src/astro/fullchart/zodiac.ts

const SIGNS = [
  "Aries","Taurus","Gemini","Cancer","Leo","Virgo",
  "Libra","Scorpio","Sagittarius","Capricorn","Aquarius","Pisces"
];

export function normalizeDeg(d: number) {
  let x = d % 360;
  if (x < 0) x += 360;
  return x;
}

export function signFromLon(lon: number) {
  const d = normalizeDeg(lon);
  const idx = Math.floor(d / 30);
  const sign = SIGNS[idx];
  const degInSign = d - idx * 30;
  return { sign, degInSign };
}

export function degToSignString(lon: number) {
  const { sign, degInSign } = signFromLon(lon);
  const deg = Math.floor(degInSign);
  const min = Math.floor((degInSign - deg) * 60);
  return `${deg}°${String(min).padStart(2, "0")}' ${sign}`;
}

6) Math + aspects

src/astro/fullchart/math.ts

import { normalizeDeg } from "./zodiac";

export function angularDistance(a: number, b: number) {
  const d = Math.abs(normalizeDeg(a) - normalizeDeg(b));
  return Math.min(d, 360 - d); // 0..180
}

export function withinOrb(distance: number, exact: number, orb: number) {
  return Math.abs(distance - exact) <= orb;
}

src/astro/fullchart/aspects.ts

import type { AspectHit, AspectKey, PlanetKey } from "./types";
import { angularDistance, withinOrb } from "./math";

const ASPECTS: { key: AspectKey; exact: number; orb: number }[] = [
  { key: "conjunction", exact: 0, orb: 8 },
  { key: "sextile", exact: 60, orb: 4 },
  { key: "square", exact: 90, orb: 6 },
  { key: "trine", exact: 120, orb: 6 },
  { key: "opposition", exact: 180, orb: 8 }
];

export function computeAspects(planets: Record<PlanetKey, { lon: number }>) {
  const keys = Object.keys(planets) as PlanetKey[];
  const hits: AspectHit[] = [];

  for (let i = 0; i < keys.length; i++) {
    for (let j = i + 1; j < keys.length; j++) {
      const a = keys[i];
      const b = keys[j];
      const dist = angularDistance(planets[a].lon, planets[b].lon);

      for (const asp of ASPECTS) {
        if (withinOrb(dist, asp.exact, asp.orb)) {
          hits.push({
            a, b,
            aspect: asp.key,
            exactDeg: asp.exact,
            orb: Math.round(Math.abs(dist - asp.exact) * 100) / 100
          });
          break;
        }
      }
    }
  }

  // Sort: tightest orb first
  hits.sort((x, y) => x.orb - y.orb);
  return hits;
}

7) Build FullChart from a provider result

This assumes your astro provider can return planet longitudes (degrees). If your current API only returns sign strings, you’ll upgrade the API endpoint later. The app is now ready for it.

src/astro/fullchart/build.ts

import type { FullChart, HouseData, PlanetKey } from "./types";
import { signFromLon } from "./zodiac";
import { computeAspects } from "./aspects";

export function assignHouse(lon: number, cusps: number[]) {
  // Very simple cusp walk. Works when cusps are ordered around the circle.
  // Return house 1..12 based on which cusp interval the lon falls into.
  const norm = (d: number) => ((d % 360) + 360) % 360;
  const L = norm(lon);

  for (let i = 0; i < 12; i++) {
    const start = norm(cusps[i]);
    const end = norm(cusps[(i + 1) % 12]);

    const inArc = start < end ? (L >= start && L < end) : (L >= start || L < end);
    if (inArc) return i + 1;
  }
  return undefined;
}

export function buildFullChart(input: {
  meta: FullChart["meta"];
  planetLon: Record<PlanetKey, number>;
  houses?: HouseData;
}) : FullChart {
  const planets: any = {};

  for (const [k, lon] of Object.entries(input.planetLon) as [PlanetKey, number][]) {
    const { sign, degInSign } = signFromLon(lon);
    const house = input.houses?.cusps?.length === 12 ? assignHouse(lon, input.houses.cusps) : undefined;

    planets[k] = {
      key: k,
      lon,
      sign,
      degInSign,
      house
    };
  }

  const aspects = computeAspects(planets);

  return {
    meta: input.meta,
    planets,
    houses: input.houses,
    aspects
  };
}

8) API endpoint: /api/astro/fullchart

This calls your provider, expects it to return planet longitudes + cusps (if time known). If not available yet, return “not implemented” cleanly.

src/app/api/astro/fullchart/route.ts

import { NextResponse } from "next/server";
import { z } from "zod";
import { getAstroProvider } from "@/astro/provider";
import { buildFullChart } from "@/astro/fullchart/build";
import type { PlanetKey } from "@/astro/fullchart/types";

const Schema = z.object({
  dateISO: z.string().regex(/^\d{4}-\d{2}-\d{2}$/),
  time24: z.string().regex(/^\d{2}:\d{2}$/).optional(),
  timeUnknown: z.boolean(),
  place: z.string().min(2),
  lat: z.number(),
  lon: z.number(),
  timezone: z.string().optional(),
  utcOffsetMin: z.number().optional(),
  isDST: z.boolean().optional(),
  houseSystem: z.enum(["placidus","whole_sign"]).optional()
});

export async function POST(req: Request) {
  try {
    const body = await req.json();
    const data = Schema.parse(body);

    const provider = getAstroProvider();
    const chart = await provider.getChart({
      dateISO: data.dateISO,
      time24: data.time24,
      timeUnknown: data.timeUnknown,
      place: data.place,
      lat: data.lat,
      lon: data.lon,
      timezone: data.timezone,
      houseSystem: data.houseSystem ?? "placidus"
    });

    // Provider must supply longitudes for Wave 2B to be truly "full"
    const lon = (chart as any).longitudes as Partial<Record<PlanetKey, number>> | undefined;
    const houses = (chart as any).houses as any;

    if (!lon || !lon.sun || !lon.moon) {
      return NextResponse.json({
        ok: false,
        error: "Provider does not return planet longitudes yet. Upgrade provider to Wave 2B spec."
      }, { status: 501 });
    }

    const full = buildFullChart({
      meta: {
        dateISO: data.dateISO,
        time24: data.time24,
        timeUnknown: data.timeUnknown,
        place: data.place,
        lat: data.lat,
        lon: data.lon,
        timezone: data.timezone,
        utcOffsetMin: data.utcOffsetMin,
        isDST: data.isDST
      },
      planetLon: lon as Record<PlanetKey, number>,
      houses: data.timeUnknown ? undefined : houses
    });

    return NextResponse.json({ ok: true, full });
  } catch (err: any) {
    return NextResponse.json({ ok: false, error: err?.message ?? "Unknown error" }, { status: 400 });
  }
}

src/poster/plot.ts

export function degToRad(deg: number) {
  return (deg * Math.PI) / 180;
}

export function polar(cx: number, cy: number, r: number, angleDeg: number) {
  // 0° at top, clockwise (poster-friendly)
  const a = degToRad(angleDeg - 90);
  return {
    x: cx + r * Math.cos(a),
    y: cy + r * Math.sin(a)
  };
}

export function lonToWheelAngle(lon: number) {
  // Map zodiac longitude directly to wheel angle
  // Aries 0° = top.
  return lon;
}

import React from "react";
import { polar, lonToWheelAngle } from "./plot";

const PLANET_GLYPHS: Record<string, string> = {

export function BirthChartPosterSVG({ data }: { data: any }) {
  const W = 1080, H = 1350;
  const cx = W / 2, cy = 610;
  const wheelR = 360;
  const planetR = 320;

  const title = data?.title ?? "Soul Codex Birth Chart";
  const date = data?.dateText ?? "";
  const time = data?.timeText ?? "";
  const place = data?.placeText ?? "";

  const lifePath = data?.lifePath ?? "";
  const lifePathTitle = data?.lifePathTitle ?? "";
  const master = data?.masterNumber ?? "";

  const planets = data?.planets ?? {}; // { sun: { lon }, ... }

  return (

      <defs>
        <radialGradient id="bg" cx="50%" cy="35%" r="75%">
          <stop offset="0%" stopColor="#0b3b3a" />
          <stop offset="55%" stopColor="#071f2a" />

          </feMerge>

      {/* Background */}
      <rect width={W} height={H} fill="url(#bg)" />
      {renderStars(140, W, H)}

        const angle = i * 30;

        const p2 = polar(cx, cy, wheelR - 40, angle);

      })}

      {/* Rim text */}

      </text>
      <text x={cx} y={cy + 98} fill="#f4f1ea" fontSize="22" textAnchor="middle" opacity="0.8" fontFamily="system-ui">

        const lon = Number(v?.lon);

function describeCirclePath(cx: number, cy: number, r: number) {
  const startX = cx;
  const startY = cy - r;
  return `M ${startX} ${startY} A ${r} ${r} 0 1 1 ${cx - 0.01} ${startY} A ${r} ${r} 0 1 1 ${startX} ${startY}`;
}

function renderStars(count: number, w: number, h: number) {
  const stars = [];
  for (let i = 0; i < count; i++) {
    const x = (i * 73) % w;
    const y = (i * 191) % h;
    const r = (i % 3) + 0.6;
    const o = 0.25 + (i % 10) / 20;
    stars.push(<circle key={i} cx={x} cy={y} r={r} fill="#fff" opacity={o} />);
  }
  return <g>{stars}</g>;
}
