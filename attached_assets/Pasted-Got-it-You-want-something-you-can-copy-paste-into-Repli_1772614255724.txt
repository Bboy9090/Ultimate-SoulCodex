```ts
// src/soulcodex/compute/numerology.ts
export function calcLifePath(dateISO: string): number {
  // dateISO: YYYY-MM-DD
  const digits = dateISO.replace(/-/g, "").split("").map(d => parseInt(d, 10));
  let sum = digits.reduce((a, b) => a + b, 0);

  // reduce to 1-9 (keep master numbers optional if you want later)
  while (sum > 9) {
    const parts = String(sum).split("").map(n => parseInt(n, 10));
    sum = parts.reduce((a, b) => a + b, 0);
  }
  return sum;
}
```

```ts
// src/soulcodex/compute/moral.ts
import type { PressureStyle } from "../types";

export function moralCode(pressureStyle: PressureStyle, nonNegotiables: string[]): { codeName: string; notes: string[] } {
  const nn = nonNegotiables.map(s => s.toLowerCase());

  const style =
    pressureStyle === "step_in" ? "Protector" :
    pressureStyle === "assess" ? "Assessor" : "Self-Preserver";

  const truthy = nn.some(x => x.includes("lie") || x.includes("dishonest"));
  const stagnation = nn.some(x => x.includes("repetit") || x.includes("stagn"));
  const disrespect = nn.some(x => x.includes("disrespect"));

  const notes: string[] = [];
  if (truthy) notes.push("Direct dishonesty is a hard boundary; trust breaks fast when truth is violated.");
  if (stagnation) notes.push("Stagnation drains you; you lose respect when people refuse to evolve or repeat the same cycle.");
  if (disrespect) notes.push("Respect is foundational; tone and conduct matter as much as intent.");

  const codeName = `${truthy ? "Truth-First" : "Values-First"} ${style}${stagnation ? " (Growth-Driven)" : ""}`;
  return { codeName, notes };
}
```

```ts
// src/soulcodex/compute/archetype.ts
import type { StressElement, PressureStyle, DecisionStyle } from "../types";
import type { Archetype } from "../types";

function roleFromDecision(decision: DecisionStyle): string {
  if (decision === "calm_logic") return "Architect";
  if (decision === "quiet_instinct") return "Oracle";
  if (decision === "willpower") return "Sovereign";
  if (decision === "sleep_on_it") return "Sage";
  return "Catalyst"; // gut_yes_no
}

function elementLabel(el: StressElement): string {
  // use as “style” not literal stress
  switch (el) {
    case "air": return "Wind";
    case "fire": return "Solar";
    case "earth": return "Stone";
    case "water": return "Ocean";
    case "metal": return "Iron";
  }
}

function pressureFlavor(ps: PressureStyle): string {
  return ps === "assess" ? "Strategist" : ps === "step_in" ? "Guardian" : "Wanderer";
}

export function chooseArchetype(stress: StressElement, pressure: PressureStyle, decision: DecisionStyle): Archetype {
  const element = elementLabel(stress);
  const role = roleFromDecision(decision);
  const flavor = pressureFlavor(pressure);

  const name = `${element} ${role}`;
  const code = `${element[0]}${role[0]}`.toUpperCase(); // simple
  const tagline =
    flavor === "Strategist"
      ? "I watch first, then move with precision."
      : flavor === "Guardian"
      ? "I protect what matters and build stability."
      : "I preserve my peace and choose my battles.";

  return { name, code, tagline };
}
```

```ts
// src/soulcodex/validators/blandnessFilter.ts

const BANNED = [
  "cosmic signature",
  "sacred blueprint",
  "divine timing",
  "vibrational frequency",
  "holistic convergence",
  "incarnation"
];

export function blandnessCheck(text: string): { ok: boolean; reasons: string[] } {
  const lower = text.toLowerCase();
  const reasons: string[] = [];

  for (const b of BANNED) {
    if (lower.includes(b)) reasons.push(`Contains banned phrase: "${b}"`);
  }

  // crude checks for missing depth signals
  const mustHave = ["when i'm stressed", "i tend to", "i'm learning to", "in relationships"];
  for (const m of mustHave) {
    if (!lower.includes(m)) reasons.push(`Missing expected phrase: "${m}"`);
  }

  return { ok: reasons.length === 0, reasons };
}
```

## 9) compute/compatibility.ts (compatibility scoring)

```ts
// src/soulcodex/compute/compatibility.ts
import type { SoulProfile } from "../types";

export interface CompatibilityScore {
  identity: number;
  stress: number;
  values: number;
  decisions: number;
  overall: number;
  summary: string[];
  friction: string[];
}

function scoreMatch(a: string, b: string, same = 90, close = 70, diff = 55): number {
  if (!a || !b) return 60;
  if (a === b) return same;
  // light heuristic: shared word
  const aw = a.split(" ");
  const bw = b.split(" ");
  if (aw.some(x => bw.includes(x))) return close;
  return diff;
}

export function compatibility(a: SoulProfile, b: SoulProfile): CompatibilityScore {
  const identity = scoreMatch(a.synthesis.archetype.name, b.synthesis.archetype.name);

  const stress = a.signals.elements.stressSignal === b.signals.elements.stressSignal ? 85 : 60;

  const valuesOverlap = a.signals.moral.nonNegotiables.filter(x =>
    b.signals.moral.nonNegotiables.map(y => y.toLowerCase()).includes(x.toLowerCase())
  ).length;

  const values = valuesOverlap >= 1 ? 85 : 65;

  const decisions = a.signals.design.decisionStyle === b.signals.design.decisionStyle ? 80 : 65;

  const overall = Math.round((identity + stress + values + decisions) / 4);

  const summary = [
    `Identity: ${identity}% (how naturally you move together)`,
    `Stress: ${stress}% (how you collide under pressure)`,
    `Values: ${values}% (trust and boundaries overlap)`,
    `Decisions: ${decisions}% (how you choose and commit)`
  ];

  const friction: string[] = [];
  if (stress < 70) friction.push("Under pressure, you regulate differently. One may need space while the other wants action.");
  if (values < 80) friction.push("Your non-negotiables differ. Without explicit boundaries, resentment builds quietly.");
  if (decisions < 75) friction.push("Decision timing may clash. Agree on how long decisions get to sit before action.");

  return { identity, stress, values, decisions, overall, summary, friction };
}
```

## 10) index.ts (the main glue you call)

```ts
// src/soulcodex/index.ts
import type { SoulProfile, UserInputs } from "./types";
import { calcLifePath } from "./compute/numerology";
import { elementNotes } from "./compute/elements";
import { moralCode } from "./compute/moral";
import { synthesize } from "./compute/synthesis";
import { buildResultsPrompt } from "./prompts/resultsEngine";
import { blandnessCheck } from "./validators/blandnessFilter";

export function buildSoulProfile(inputs: UserInputs): { profile: SoulProfile; prompt: string; validation: ReturnType<typeof blandnessCheck> } {
  const lifePath = inputs.numerology?.lifePath ?? calcLifePath(inputs.birth.dateISO);

  const moral = moralCode(inputs.pressureStyle, inputs.nonNegotiables);
  const elNotes = elementNotes(inputs.stressElement);

  const profile: SoulProfile = {
    user: {
      displayName: inputs.displayName,
      codename: inputs.codename,
      birth: inputs.birth
    },
    signals: {
      astrology: {
        sun: inputs.astrology?.sun,
        moon: inputs.astrology?.moon,
        rising: inputs.astrology?.rising,
        notes: inputs.birth.timeUnknown ? ["Birth time unknown: Rising/Houses may vary."] : []
      },
      numerology: {
        lifePath,
        expression: inputs.numerology?.expression,
        soulUrge: inputs.numerology?.soulUrge,
        personality: inputs.numerology?.personality,
        notes: ["Life Path computed from birth date."]
      },
      elements: {
        stressSignal: inputs.stressElement,
        notes: elNotes
      },
      moral: {
        pressureStyle: inputs.pressureStyle,
        nonNegotiables: inputs.nonNegotiables,
        codeName: moral.codeName,
        notes: moral.notes
      },
      design: {
        decisionStyle: inputs.decisionStyle,
        socialEnergy: inputs.socialEnergy,
        notes: ["Decision style and social energy are behavioral proxies unless bodygraph is computed."]
      }
    },
    synthesis: {} as any,
    outputs: {}
  };

  profile.synthesis = synthesize(profile.signals);

  const prompt = buildResultsPrompt(profile, inputs.depthMode);

  // validation is meant for the generated LLM output, but we can use it on prompts too
  const validation = blandnessCheck(prompt);

  return { profile, prompt, validation };
}
```

# How you use this in Replit/Cursor (simple)

* You call `buildSoulProfile(inputs)`
* It returns:

  * `profile` (structured)
  * `prompt` (send to your LLM)
  * `validation` (rules check)

Then your backend calls OpenAI/LLM with that prompt and stores result in `profile.outputs[mode]`.
