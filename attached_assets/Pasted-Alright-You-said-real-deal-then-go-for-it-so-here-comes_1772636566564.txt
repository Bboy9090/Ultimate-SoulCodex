Wave 3.5 “30-System Synthesis Engine” Architecture

Core idea

Instead of generating one big horoscope text, we generate:
	1.	Signals (facts + probabilities + intensity)
	2.	Themes (clusters like “precision,” “privacy,” “service,” “power,” “freedom”)
	3.	Claims (short, testable statements about the user)
	4.	Narrative (first-person output that references their signals, not definitions)
	5.	Prescriptions (what to do this week, what not to do, what to watch for)

Two-layer brain
	•	Deterministic layer (rules/math): stable, repeatable, debuggable
	•	Narrative layer (LLM): makes it human, poetic, not robotic
	•	Quality gate: if generic → rewrite with stronger anchors

This keeps your app from turning into a state exam while still being specific.

3) Cluster + score engine

src/codex30/synth/score.ts

import type { Signal, ThemeScore } from "../types";

export function scoreThemes(signals: Signal[]): ThemeScore[] {
  const map = new Map<string, { score: number; sources: string[] }>();

  for (const s of signals) {
    for (const tag of s.tags) {
      const current = map.get(tag) ?? { score: 0, sources: [] };
      // Basic scoring: intensity * confidence weight * polarity
      const confW = s.confidence === "high" ? 1 : s.confidence === "medium" ? 0.7 : 0.45;
      const polW = s.polarity === "strength" ? 1 : s.polarity === "shadow" ? 0.8 : 0.9;

      current.score += 100 * s.intensity * confW * polW;
      current.sources.push(s.id);

      map.set(tag, current);
    }
  }

  const scored = Array.from(map.entries()).map(([tag, v]) => ({
    tag,
    score: Math.round(v.score),
    sources: Array.from(new Set(v.sources))
  }));

  scored.sort((a, b) => b.score - a.score);
  return scored.slice(0, 12);
}

import type { CodexSynthesis, Signal, ThemeScore } from "../types";

export function compileBulletLists(signals: Signal[], themes: ThemeScore[]) {
  const strengths: string[] = [];
  const shadows: string[] = [];
  const triggers: string[] = [];
  const prescriptions: string[] = [];

  // Pull top signals with strength/shadow polarity
  const topStrength = signals
    .filter(s => s.polarity === "strength")
    .sort((a,b) => b.intensity - a.intensity)
    .slice(0, 6);

  const topShadow = signals
    .filter(s => s.polarity === "shadow")
    .sort((a,b) => b.intensity - a.intensity)
    .slice(0, 6);

  for (const s of topStrength) strengths.push(s.label);
  for (const s of topShadow) shadows.push(s.label);

  // Triggers + prescriptions come from theme tags
  const tags = themes.slice(0, 6).map(t => t.tag);

  if (tags.includes("precision")) triggers.push("Sloppy work, vague promises, and people who won’t follow through.");
  if (tags.includes("privacy")) triggers.push("Intrusion, forced openness, and being watched while you’re building.");
  if (tags.includes("truth")) triggers.push("Lies to your face, manipulation, and ‘word salad’ excuses.");
  if (tags.includes("social_sensitivity")) triggers.push("Noisy environments and chaotic people when you’re already full.");

  // Prescriptions are “do this this week”
  prescriptions.push("Pick one build target and finish it. Your nervous system trusts completion.");
  prescriptions.push("Reduce input. Less scrolling, more crafting. Information is not progress.");
  prescriptions.push("Say ‘no’ once this week without explaining yourself.");

  return { strengths, shadows, triggers, prescriptions };
}

export function pickCodename(themes: ThemeScore[]) {
  const top = themes[0]?.tag ?? "legacy";
  const map: Record<string, string> = {
    precision: "The Quiet Blade Architect",
    legacy: "The Immortal Forge Builder",
    privacy: "The Hidden Lighthouse",
    intensity: "The Calm Volcano",
    intuition: "The Signal Reader"
  };
  return map[top] ?? "The Quiet Storm Architect";
}

export function isGeneric(text: string) {
  const t = text.toLowerCase();

  const genericPhrases = [
    "you are a unique individual",
    "your journey",
    "the universe",
    "cosmic blueprint",
    "embrace your truth",
    "growth and transformation",
    "aligned with your purpose"
  ];

  const hits = genericPhrases.filter(p => t.includes(p)).length;

  // Also require specific anchors like signs/values/non-negotiables/goals
  const hasConcrete =
    /(virgo|scorpio|libra|aries|capricorn|aquarius|leo|taurus|gemini|cancer|pisces|sagittarius)/i.test(text) ||
    /(lies|repetitiveness|boundaries|legacy|immortal|build)/i.test(text);

  return hits >= 2 || !hasConcrete || text.length < 600;
}

import type { Signal, SystemId } from "../types";

export function astrologySignals(fullChart: any, verified: boolean): Signal[] {
  const system: SystemId = "astrology";
  const conf = verified ? "high" : "medium";

  const sun = fullChart?.planets?.sun?.sign;
  const moon = fullChart?.planets?.moon?.sign;
  const rising = fullChart?.planets?.rising?.sign || fullChart?.meta?.timeUnknown ? undefined : fullChart?.houses?.ascSign;

  const out: Signal[] = [];

  if (sun) out.push({
    id: `astro.sun.${sun.toLowerCase()}`,
    system,
    label: `Identity expresses through ${sun} patterns (how you build, decide, and judge quality).`,
    evidence: [`Sun in ${sun}`],
    intensity: 0.9,
    polarity: "neutral",
    confidence: conf as any,
    tags: sun === "Virgo" ? ["precision","craft","service","order"] : ["craft","legacy"]
  });

  if (moon) out.push({
    id: `astro.moon.${moon.toLowerCase()}`,
    system,
    label: `Emotional processing follows ${moon} patterns (what soothes you, what sets you off).`,
    evidence: [`Moon in ${moon}`],
    intensity: 0.8,
    polarity: "neutral",
    confidence: conf as any,
    tags: moon === "Virgo" ? ["precision","order","discipline"] : ["emotion_depth","intuition"]
  });

  if (rising) out.push({
    id: `astro.rising.${String(rising).toLowerCase()}`,
    system,
    label: `First impression and defense style reflect ${rising} energy (how you guard and approach).`,
    evidence: [`Rising in ${rising}`],
    intensity: 0.75,
    polarity: "neutral",
    confidence: "high",
    tags: String(rising) === "Scorpio" ? ["privacy","intensity","boundaries","truth"] : ["boundaries","leadership"]
  });

  return out;
}

export function aspectSignals(fullChart: any): Signal[] {
  const aspects = fullChart?.aspects ?? [];
  const out: Signal[] = [];

  // Take top tight aspects
  for (const a of aspects.slice(0, 6)) {
    const label = `${a.a} ${a.aspect} ${a.b} (orb ${a.orb}°) shapes how your inner systems cooperate or clash.`;
    const tags =
      a.aspect === "square" ? ["intensity","discipline","courage"] :
      a.aspect === "opposition" ? ["boundaries","truth","intensity"] :
      a.aspect === "trine" ? ["intuition","healing","craft"] :
      a.aspect === "conjunction" ? ["focus","legacy","intensity"] :
      ["innovation","freedom"];

    out.push({
      id: `aspect.${a.a}.${a.aspect}.${a.b}`,
      system: "aspects",
      label,
      evidence: [`${a.a} ${a.aspect} ${a.b}`],
      intensity: Math.max(0.55, 1 - (a.orb / 10)),
      polarity: a.aspect === "square" ? "shadow" : "strength",
      confidence: "high",
      tags
    });
  }

  return out;
}

      polarity: "strength",
      confidence: "high",

import type { Codex30Input, Signal } from "./types";
import { astrologySignals } from "./systems/astrology";
import { aspectSignals } from "./systems/aspects";
import { numerologySignals } from "./systems/numerology";
import { humanDesignSignals } from "./systems/humanDesign";
import { elementSignals } from "./systems/elements";
import { moralCompassSignals } from "./systems/moralCompass";

export function collectSignals(input: Codex30Input): Signal[] {
  const verified = input?.profile?.meta?.confidence?.badge === "verified";

  const s: Signal[] = [
    ...astrologySignals(input.fullChart, verified),
    ...aspectSignals(input.fullChart),
    ...numerologySignals(input.profile?.signals ?? input.profile ?? {}),
    ...humanDesignSignals(input.profile?.signals?.humanDesign ?? input.profile?.humanDesign),
    ...elementSignals(input.userInputs),
    ...moralCompassSignals(input.userInputs)
  ];

  // Remove duplicates by id
  const map = new Map<string, Signal>();
  for (const x of s) map.set(x.id, x);
  return Array.from(map.values());
}

8) Generator endpoint (Codex30 output)

src/app/api/codex30/generate/route.ts

import { NextResponse } from "next/server";
import { z } from "zod";
import { collectSignals } from "@/codex30/registry";
import { scoreThemes } from "@/codex30/synth/score";
import { compileBulletLists, pickCodename } from "@/codex30/synth/compile";
import { isGeneric } from "@/codex30/synth/quality";
import { narratorPrompt } from "@/codex30/prompts/narrator";
import { rewritePrompt } from "@/codex30/prompts/rewrite";
import { getOpenAIConfig } from "@/lib/openai";

const Schema = z.object({
  profile: z.any(),
  fullChart: z.any().optional(),
  userInputs: z.any()
});

async function callOpenAI(input: string) {
  const { apiKey, model } = getOpenAIConfig();

  const res = await fetch("https://api.openai.com/v1/responses", {
    method: "POST",
    headers: {
      "Authorization": `Bearer ${apiKey}`,
      "Content-Type": "application/json"
    },
    body: JSON.stringify({ model, input })
  });

  if (!res.ok) throw new Error(`OpenAI error: ${res.status} ${await res.text()}`);
  const data = await res.json();

  const text =
    data?.output_text ??
    data?.output?.[0]?.content?.find((c: any) => c.type === "output_text")?.text ??
    data?.output?.[0]?.content?.[0]?.text ??
    "";

  if (!text) throw new Error("Empty model output");
  return text;
}

export async function POST(req: Request) {
  try {
    const body = await req.json();
    const input = Schema.parse(body);

    const signals = collectSignals(input);
    const themes = scoreThemes(signals);
    const codename = pickCodename(themes);

    // Basic archetype name (you can expand later with a real archetype library)
    const archetype = `${codename}`;

    const { strengths, shadows, triggers, prescriptions } = compileBulletLists(signals, themes);

    const anchors = signals
      .flatMap(s => s.evidence)
      .filter(Boolean)
      .slice(0, 12);

    let narrative = await callOpenAI(narratorPrompt({
      codename,
      archetype,
      themes: themes.slice(0, 6).map(t => ({ tag: t.tag, score: t.score })),
      strengths,
      shadows,
      triggers,
      prescriptions,
      anchors
    }));

    // Quality gate rewrite
    if (isGeneric(narrative)) {
      narrative = await callOpenAI(rewritePrompt(narrative, anchors));
    }

    const confidence = input.profile?.meta?.confidence
      ? { confidenceLabel: input.profile.meta.confidence.label, reason: input.profile.meta.confidence.reason }
      : { confidenceLabel: "Unverified", reason: "No confidence data." };

    return NextResponse.json({
      ok: true,
      synthesis: {
        codename,
        archetype,
        badges: confidence,
        topThemes: themes,
        strengths,
        shadows,
        triggers,
        prescriptions,
        narrative,
        debug: { signals, themeScores: themes }
      }
    });
  } catch (err: any) {
    return NextResponse.json({ ok: false, error: err?.message ?? "Unknown error" }, { status: 400 });
  }
}
