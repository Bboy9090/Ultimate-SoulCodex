// src/soulcodex/compute/numerology.ts
export function calcLifePath(dateISO: string): number {
  // dateISO: YYYY-MM-DD
  const digits = dateISO.replace(/-/g, "").split("").map(d => parseInt(d, 10));
  let sum = digits.reduce((a, b) => a + b, 0);

  // reduce to 1-9 (keep master numbers optional if you want later)
  while (sum > 9) {
    const parts = String(sum).split("").map(n => parseInt(n, 10));
    sum = parts.reduce((a, b) => a + b, 0);
  }
  return sum;
}

import type { PressureStyle } from "../types";

export function moralCode(pressureStyle: PressureStyle, nonNegotiables: string[]): { codeName: string; notes: string[] } {
  const nn = nonNegotiables.map(s => s.toLowerCase());

  const style =
    pressureStyle === "step_in" ? "Protector" :
    pressureStyle === "assess" ? "Assessor" : "Self-Preserver";

  const truthy = nn.some(x => x.includes("lie") || x.includes("dishonest"));
  const stagnation = nn.some(x => x.includes("repetit") || x.includes("stagn"));
  const disrespect = nn.some(x => x.includes("disrespect"));

  const notes: string[] = [];
  if (truthy) notes.push("Direct dishonesty is a hard boundary; trust breaks fast when truth is violated.");
  if (stagnation) notes.push("Stagnation drains you; you lose respect when people refuse to evolve or repeat the same cycle.");
  if (disrespect) notes.push("Respect is foundational; tone and conduct matter as much as intent.");

  const codeName = `${truthy ? "Truth-First" : "Values-First"} ${style}${stagnation ? " (Growth-Driven)" : ""}`;
  return { codeName, notes };
}

// src/soulcodex/compute/archetype.ts
import type { StressElement, PressureStyle, DecisionStyle } from "../types";
import type { Archetype } from "../types";

function roleFromDecision(decision: DecisionStyle): string {
  if (decision === "calm_logic") return "Architect";
  if (decision === "quiet_instinct") return "Oracle";
  if (decision === "willpower") return "Sovereign";
  if (decision === "sleep_on_it") return "Sage";
  return "Catalyst"; // gut_yes_no
}

function elementLabel(el: StressElement): string {
  // use as “style” not literal stress
  switch (el) {
    case "air": return "Wind";
    case "fire": return "Solar";
    case "earth": return "Stone";
    case "water": return "Ocean";
    case "metal": return "Iron";
  }
}

function pressureFlavor(ps: PressureStyle): string {
  return ps === "assess" ? "Strategist" : ps === "step_in" ? "Guardian" : "Wanderer";
}

export function chooseArchetype(stress: StressElement, pressure: PressureStyle, decision: DecisionStyle): Archetype {
  const element = elementLabel(stress);
  const role = roleFromDecision(decision);
  const flavor = pressureFlavor(pressure);

  const name = `${element} ${role}`;
  const code = `${element[0]}${role[0]}`.toUpperCase(); // simple
  const tagline =
    flavor === "Strategist"
      ? "I watch first, then move with precision."
      : flavor === "Guardian"
      ? "I protect what matters and build stability."
      : "I preserve my peace and choose my battles.";

  return { name, code, tagline };
}

## 8) validators/blandnessFilter.ts (simple post-check)

```ts
// src/soulcodex/validators/blandnessFilter.ts

const BANNED = [
  "cosmic signature",
  "sacred blueprint",
  "divine timing",
  "vibrational frequency",
  "holistic convergence",
  "incarnation"
];

export function blandnessCheck(text: string): { ok: boolean; reasons: string[] } {
  const lower = text.toLowerCase();
  const reasons: string[] = [];

  for (const b of BANNED) {
    if (lower.includes(b)) reasons.push(`Contains banned phrase: "${b}"`);
  }

  // crude checks for missing depth signals
  const mustHave = ["when i'm stressed", "i tend to", "i'm learning to", "in relationships"];
  for (const m of mustHave) {
    if (!lower.includes(m)) reasons.push(`Missing expected phrase: "${m}"`);
  }

  return { ok: reasons.length === 0, reasons };
}
```

## 9) compute/compatibility.ts (compatibility scoring)

```ts
// src/soulcodex/compute/compatibility.ts
import type { SoulProfile } from "../types";

export interface CompatibilityScore {
  identity: number;
  stress: number;
  values: number;
  decisions: number;
  overall: number;
  summary: string[];
  friction: string[];
}

function scoreMatch(a: string, b: string, same = 90, close = 70, diff = 55): number {
  if (!a || !b) return 60;
  if (a === b) return same;
  // light heuristic: shared word
  const aw = a.split(" ");
  const bw = b.split(" ");
  if (aw.some(x => bw.includes(x))) return close;
  return diff;
}

export function compatibility(a: SoulProfile, b: SoulProfile): CompatibilityScore {
  const identity = scoreMatch(a.synthesis.archetype.name, b.synthesis.archetype.name);

  const stress = a.signals.elements.stressSignal === b.signals.elements.stressSignal ? 85 : 60;

  const valuesOverlap = a.signals.moral.nonNegotiables.filter(x =>
    b.signals.moral.nonNegotiables.map(y => y.toLowerCase()).includes(x.toLowerCase())
  ).length;

  const values = valuesOverlap >= 1 ? 85 : 65;

  const decisions = a.signals.design.decisionStyle === b.signals.design.decisionStyle ? 80 : 65;

  const overall = Math.round((identity + stress + values + decisions) / 4);

  const summary = [
    `Identity: ${identity}% (how naturally you move together)`,
    `Stress: ${stress}% (how you collide under pressure)`,
    `Values: ${values}% (trust and boundaries overlap)`,
    `Decisions: ${decisions}% (how you choose and commit)`
  ];

  const friction: string[] = [];
  if (stress < 70) friction.push("Under pressure, you regulate differently. One may need space while the other wants action.");
  if (values < 80) friction.push("Your non-negotiables differ. Without explicit boundaries, resentment builds quietly.");
  if (decisions < 75) friction.push("Decision timing may clash. Agree on how long decisions get to sit before action.");

  return { identity, stress, values, decisions, overall, summary, friction };
}
```

import type { SoulProfile, UserInputs } from "./types";
import { calcLifePath } from "./compute/numerology";

export function buildSoulProfile(inputs: UserInputs): { profile: SoulProfile; prompt: string; validation: ReturnType<typeof blandnessCheck> } {
  const lifePath = inputs.numerology?.lifePath ?? calcLifePath(inputs.birth.dateISO);

  const profile: SoulProfile = {

    },
    signals: {
      astrology: {
        sun: inputs.astrology?.sun,
        moon: inputs.astrology?.moon,
        rising: inputs.astrology?.rising,
        notes: inputs.birth.timeUnknown ? ["Birth time unknown: Rising/Houses may vary."] : []
      },
      numerology: {
        lifePath,
        expression: inputs.numerology?.expression,
        soulUrge: inputs.numerology?.soulUrge,
        personality: inputs.numerology?.personality,
        notes: ["Life Path computed from birth date."]
      },
      elements: {
        stressSignal: inputs.stressElement,
        notes: elNotes
      },
      moral: {
        pressureStyle: inputs.pressureStyle,
        nonNegotiables: inputs.nonNegotiables,
        codeName: moral.codeName,
        notes: moral.notes
      },
      design: {
        decisionStyle: inputs.decisionStyle,
        socialEnergy: inputs.socialEnergy,
        notes: ["Decision style and social energy are behavioral proxies unless bodygraph is computed."]
      }
    },
    synthesis: {} as any,
    outputs: {}
  };

  profile.synthesis = synthesize(profile.signals);

  const prompt = buildResultsPrompt(profile, inputs.depthMode);

  // validation is meant for the generated LLM output, but we can use it on prompts too
  const validation = blandnessCheck(prompt);

  return { profile, prompt, validation };
}

## 4) API Route: Generate Soul Codex

### `src/app/api/soulcodex/generate/route.ts`

```ts
import { NextResponse } from "next/server";
import { z } from "zod";

import { buildSoulProfile } from "@/soulcodex";
import type { UserInputs } from "@/soulcodex/types";
import { blandnessCheck } from "@/soulcodex/validators/blandnessFilter";
import { getOpenAIConfig } from "@/lib/openai";

const InputsSchema = z.object({
  displayName: z.string().optional(),
  codename: z.string().optional(),
  birth: z.object({
    dateISO: z.string().regex(/^\d{4}-\d{2}-\d{2}$/),
    time24: z.string().regex(/^\d{2}:\d{2}$/).optional(),
    timeUnknown: z.boolean(),
    place: z.string().min(2),
    timezone: z.string().optional()
  }),

  astrology: z.object({
    sun: z.string().optional(),
    moon: z.string().optional(),
    rising: z.string().optional()
  }).optional(),

  numerology: z.object({
    lifePath: z.number().int().min(1).max(9).optional(),
    expression: z.number().int().min(1).max(9).optional(),
    soulUrge: z.number().int().min(1).max(9).optional(),
    personality: z.number().int().min(1).max(9).optional()
  }).optional(),

  stressElement: z.enum(["earth", "water", "fire", "air", "metal"]),
  pressureStyle: z.enum(["step_in", "assess", "exit"]),
  nonNegotiables: z.array(z.string()).min(1).max(3),
  decisionStyle: z.enum(["sleep_on_it", "quiet_instinct", "calm_logic", "willpower", "gut_yes_no"]),
  socialEnergy: z.enum(["steady", "bursts", "sensitive"]),
  goals: z.array(z.string()).min(1).max(2),
  depthMode: z.enum(["snapshot", "deep", "surgical"])
});

async function callOpenAI(prompt: string) {
  const { apiKey, model } = getOpenAIConfig();

  // Minimal OpenAI-compatible call using fetch.
  // If you already use the official SDK, swap this out.
  const res = await fetch("https://api.openai.com/v1/responses", {
    method: "POST",
    headers: {
      "Authorization": `Bearer ${apiKey}`,
      "Content-Type": "application/json"
    },
    body: JSON.stringify({
      model,
      input: prompt
    })
  });

  if (!res.ok) {
    const txt = await res.text();
    throw new Error(`OpenAI error: ${res.status} ${txt}`);
  }

  const data = await res.json();

  // Responses API returns content in different shapes depending on config.
  // This tries common patterns without pretending it’s perfect.
  const text =
    data?.output_text ??
    data?.output?.[0]?.content?.find((c: any) => c.type === "output_text")?.text ??
    data?.output?.[0]?.content?.[0]?.text ??
    "";

  if (!text) throw new Error("OpenAI returned empty text.");

  return text;
}

export async function POST(req: Request) {
  try {
    const body = await req.json();
    const parsed = InputsSchema.parse(body) as UserInputs;

    // Build deterministic profile + prompt
    const { profile, prompt } = buildSoulProfile(parsed);

    // Call LLM
    const output = await callOpenAI(prompt);

    // Post-check blandness (run on output, not prompt)
    const validation = blandnessCheck(output);

    // Store output in profile
    profile.outputs[parsed.depthMode] = output;

    return NextResponse.json({
      ok: true,
      profile,
      output,
      validation
    });
  } catch (err: any) {
    return NextResponse.json(
      { ok: false, error: err?.message ?? "Unknown error" },
      { status: 400 }
    );
  }
}
```

**What this gives you:** send inputs → get back `profile + output + validation`.

## 5) API Route: Compatibility

import { NextResponse } from "next/server";
import { z } from "zod";

import { buildSoulProfile } from "@/soulcodex";
import type { UserInputs } from "@/soulcodex/types";
import { compatibility } from "@/soulcodex/compute/compatibility";

const PayloadSchema = z.object({
  a: z.any(),
  b: z.any()
});

export async function POST(req: Request) {
  try {
    const body = await req.json();
    const { a, b } = PayloadSchema.parse(body) as { a: UserInputs; b: UserInputs };

    const builtA = buildSoulProfile(a).profile;
    const builtB = buildSoulProfile(b).profile;

    const score = compatibility(builtA, builtB);

    return NextResponse.json({
      ok: true,
      a: builtA,
      b: builtB,
      score
    });
  } catch (err: any) {
    return NextResponse.json(
      { ok: false, error: err?.message ?? "Unknown error" },
      { status: 400 }
    );
  }
}

  const [socialEnergy, setSocialEnergy] = useState<SocialEnergy>("sensitive");
  const [depthMode, setDepthMode] = useState<DepthMode>("deep");

  const [nonNegotiables, setNonNegotiables] = useState<string>("lies, repetitiveness");
  const [goals, setGoals] = useState<string>("self healing, build something immortal");

  const payload = useMemo(() => {
    const nn = nonNegotiables
      .split(",")
      .map(s => s.trim())
      .filter(Boolean)
      .slice(0, 3);

    const gs = goals
      .split(",")
      .map(s => s.trim())
      .filter(Boolean)
      .slice(0, 2);

    return {
      displayName,
      birth: {
        dateISO,
        time24: timeUnknown ? undefined : time24,
        timeUnknown,
        place,
        timezone
      },
      stressElement,
      pressureStyle,
      nonNegotiables: nn,
      decisionStyle,
      socialEnergy,
      goals: gs,
      depthMode
    };
  }, [
    displayName, dateISO, timeUnknown, time24, place, timezone,
    stressElement, pressureStyle, nonNegotiables,
    decisionStyle, socialEnergy, goals, depthMode
  ]);

  async function generate() {
    setLoading(true);
    setError("");
    setOutput("");
    setValidation(null);

    try {
      const res = await fetch("/api/soulcodex/generate", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify(payload)
      });

      const data = await res.json();
      if (!data.ok) throw new Error(data.error || "Request failed");

      setOutput(data.output);
      setValidation(data.validation);
    } catch (e: any) {
      setError(e?.message || "Something went wrong");
    } finally {
      setLoading(false);
    }
  }

  return (
    <div style={{ maxWidth: 900, margin: "40px auto", padding: 16, fontFamily: "system-ui" }}>
      <h1 style={{ fontSize: 28, marginBottom: 8 }}>Soul Codex</h1>
      <p style={{ opacity: 0.75, marginTop: 0 }}>
        Minimal generator. Pretty UI comes later. Function first, vanity second.
      </p>

      <div style={{ display: "grid", gridTemplateColumns: "1fr 1fr", gap: 12 }}>
        <label>
          Name
          <input value={displayName} onChange={(e) => setDisplayName(e.target.value)} style={{ width: "100%" }} />
        </label>

        <label>
          Depth
          <select value={depthMode} onChange={(e) => setDepthMode(e.target.value as any)} style={{ width: "100%" }}>
            <option value="snapshot">Snapshot</option>
            <option value="deep">Deep</option>
            <option value="surgical">Surgical</option>
          </select>
        </label>

        <label>
          Birth Date (YYYY-MM-DD)
          <input value={dateISO} onChange={(e) => setDateISO(e.target.value)} style={{ width: "100%" }} />
        </label>

        <label>
          Birth Time Unknown?
          <select value={String(timeUnknown)} onChange={(e) => setTimeUnknown(e.target.value === "true")} style={{ width: "100%" }}>
            <option value="false">No</option>
            <option value="true">Yes</option>
          </select>
        </label>

        <label>
          Birth Time (HH:MM)
          <input value={time24} onChange={(e) => setTime24(e.target.value)} style={{ width: "100%" }} disabled={timeUnknown} />
        </label>

        <label>
          Birth Place
          <input value={place} onChange={(e) => setPlace(e.target.value)} style={{ width: "100%" }} />
        </label>

        <label>
          Timezone
          <input value={timezone} onChange={(e) => setTimezone(e.target.value)} style={{ width: "100%" }} />
        </label>

        <label>
          Stress Element
          <select value={stressElement} onChange={(e) => setStressElement(e.target.value as any)} style={{ width: "100%" }}>
            <option value="air">Air</option>
            <option value="fire">Fire</option>
            <option value="earth">Earth</option>
            <option value="water">Water</option>
            <option value="metal">Metal</option>
          </select>
        </label>

        <label>
          Crisis Style
          <select value={pressureStyle} onChange={(e) => setPressureStyle(e.target.value as any)} style={{ width: "100%" }}>

        <label>
          Social Energy

            <option value="steady">Steady</option>

        <label style={{ gridColumn: "1 / -1" }}>
          Non-negotiables (comma-separated, max 3)
          <input value={nonNegotiables} onChange={(e) => setNonNegotiables(e.target.value)} style={{ width: "100%" }} />

        <label style={{ gridColumn: "1 / -1" }}>
          Goals (comma-separated, max 2)
          <input value={goals} onChange={(e) => setGoals(e.target.value)} style={{ width: "100%" }} />
        </label>
      </div>

      <button onClick={generate} disabled={loading} style={{ marginTop: 16, padding: "10px 14px" }}>
        {loading ? "Generating..." : "Generate Soul Codex"}
      </button>

      {error && <p style={{ color: "crimson" }}>{error}</p>}

      {validation && (
        <div style={{ marginTop: 12, padding: 12, border: "1px solid #ddd" }}>
          <strong>Quality Check:</strong>{" "}
          {validation.ok ? "✅ Passed" : "⚠️ Needs rewrite"}
          {!validation.ok && (
            <ul>
              {validation.reasons?.map((r: string, i: number) => <li key={i}>{r}</li>)}
            </ul>
          )}
        </div>
      )}

      {output && (
        <div style={{ marginTop: 16, padding: 16, border: "1px solid #ddd", whiteSpace: "pre-wrap" }}>
          {output}
        </div>
      )}
    </div>
  );
}
