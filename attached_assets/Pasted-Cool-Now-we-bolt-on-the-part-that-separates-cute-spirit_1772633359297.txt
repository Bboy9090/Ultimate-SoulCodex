import { prisma } from "@/lib/prisma";
import type { AstroResult } from "./types";

export async function getCachedAstro(key: string): Promise<AstroResult | null> {
  const row = await prisma.astroCache.findUnique({ where: { key } });
  return row?.payload as any ?? null;
}

export async function setCachedAstro(key: string, payload: AstroResult): Promise<void> {
  await prisma.astroCache.upsert({
    where: { key },
    update: { payload: payload as any },
    create: { key, payload: payload as any }
  });
}

4) Key builder (consistent cache key)

src/astro/key.ts

import type { AstroRequest } from "./types";

export function astroCacheKey(req: AstroRequest) {
  // Build a deterministic key. If timeUnknown, rising/houses are not computed.
  const hs = req.houseSystem ?? "placidus";
  const t = req.timeUnknown ? "time_unknown" : (req.time24 ?? "00:00");
  const lat = req.lat?.toFixed(5) ?? "na";
  const lon = req.lon?.toFixed(5) ?? "na";
  const tz = req.timezone ?? "na";
  const place = req.place.trim().toLowerCase();

  return `astro:v1:${req.dateISO}:${t}:${tz}:${place}:${lat}:${lon}:${hs}`;
}

5) Provider wrapper that always uses cache

src/astro/cachedProvider.ts

import type { AstroProvider, AstroRequest, AstroResult } from "./types";
import { astroCacheKey } from "./key";
import { getCachedAstro, setCachedAstro } from "./cache";

export function withAstroCache(provider: AstroProvider): AstroProvider {
  return {
    name: `${provider.name}+cache`,
    async getChart(req: AstroRequest): Promise<AstroResult> {
      const key = astroCacheKey(req);
      const cached = await getCachedAstro(key);
      if (cached) return cached;

      const fresh = await provider.getChart(req);
      await setCachedAstro(key, fresh);
      return fresh;
    }
  };
}

import type { AstroProvider, AstroRequest, AstroResult } from "../types";

// This is a generic provider scaffold.
// Plug in whichever API you choose by adjusting the URL + response parsing.
export function makeApiAstroProvider(opts: {
  name: string;
  baseUrl: string;        // e.g. "https://your-astro-api.com"
  apiKey?: string;
}): AstroProvider {
  return {
    name: opts.name,
    async getChart(req: AstroRequest): Promise<AstroResult> {
      if (!req.dateISO) throw new Error("Missing dateISO");
      if (!req.place) throw new Error("Missing place");

      // If you don't have lat/lon, your API might accept place text.
      // If it requires lat/lon, add geocoding step (I can give that next).
      const payload = {
        date: req.dateISO,
        time: req.timeUnknown ? null : (req.time24 ?? "00:00"),
        place: req.place,
        timezone: req.timezone ?? null,
        lat: req.lat ?? null,
        lon: req.lon ?? null,
        houseSystem: req.houseSystem ?? "placidus"
      };

      const res = await fetch(`${opts.baseUrl}/chart`, {
        method: "POST",
        headers: {
          "Content-Type": "application/json",
          ...(opts.apiKey ? { "Authorization": `Bearer ${opts.apiKey}` } : {})
        },
        body: JSON.stringify(payload)
      });

      if (!res.ok) throw new Error(`Astro API error: ${res.status} ${await res.text()}`);
      const data = await res.json();

      // Adapt this mapping to your API’s exact response.
      const out: AstroResult = {
        sun: data.sunSign,
        moon: data.moonSign,
        rising: req.timeUnknown ? undefined : data.risingSign,
        planets: data.planets ?? undefined,
        houses: data.houses ? { system: payload.houseSystem, cusps: data.houses } : undefined,
        notes: []
      };

      if (req.timeUnknown) out.notes?.push("Birth time unknown: Rising/Houses omitted.");
      return out;
    }
  };
}

8) Provider selection (single switch)

src/astro/provider.ts

import type { AstroProvider } from "./types";
import { withAstroCache } from "./cachedProvider";
import { makeApiAstroProvider } from "./providers/apiProvider";
import { localSwissEphemerisProvider } from "./providers/localProvider";

export function getAstroProvider(): AstroProvider {
  const mode = process.env.ASTRO_PROVIDER ?? "api"; // "api" | "local"

  if (mode === "local") {
    return withAstroCache(localSwissEphemerisProvider);
  }

  const baseUrl = process.env.ASTRO_API_BASE_URL;
  if (!baseUrl) throw new Error("Missing ASTRO_API_BASE_URL (set ASTRO_PROVIDER=local or configure API).");

  const apiKey = process.env.ASTRO_API_KEY;

  const api = makeApiAstroProvider({
    name: "astro-api",
    baseUrl,
    apiKey
  });

  return withAstroCache(api);
}

src/app/api/astro/chart/route.ts

import { NextResponse } from "next/server";
import { z } from "zod";
import { getAstroProvider } from "@/astro/provider";

const Schema = z.object({
  dateISO: z.string().regex(/^\d{4}-\d{2}-\d{2}$/),
  time24: z.string().regex(/^\d{2}:\d{2}$/).optional(),
  timeUnknown: z.boolean(),
  place: z.string().min(2),
  timezone: z.string().optional(),
  houseSystem: z.enum(["placidus", "whole_sign"]).optional()
});

export async function POST(req: Request) {
  try {
    const body = await req.json();
    const data = Schema.parse(body);

    const provider = getAstroProvider();
    const chart = await provider.getChart(data);

    return NextResponse.json({ ok: true, provider: provider.name, chart });
  } catch (err: any) {
    return NextResponse.json({ ok: false, error: err?.message ?? "Unknown error" }, { status: 400 });
  }
}
