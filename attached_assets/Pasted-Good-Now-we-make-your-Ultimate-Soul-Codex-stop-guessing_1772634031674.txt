3) Geocoding layer (free, cached)

src/geo/normalize.ts

export function normalizePlaceQuery(q: string) {
  return q.trim().toLowerCase().replace(/\s+/g, " ");
}

export function geoCacheKey(q: string) {
  return `geo:v1:${normalizePlaceQuery(q)}`;
}

src/geo/cache.ts

import { prisma } from "@/lib/prisma";
import { geoCacheKey, normalizePlaceQuery } from "./normalize";

export async function getGeoCached(query: string) {
  const key = geoCacheKey(query);
  return prisma.geoCache.findUnique({ where: { key } });
}

export async function setGeoCached(input: {
  query: string;
  normalizedPlace: string;
  lat: number;
  lon: number;
  provider: string;
}) {
  const key = geoCacheKey(input.query);
  return prisma.geoCache.upsert({
    where: { key },
    update: {
      query: normalizePlaceQuery(input.query),
      normalizedPlace: input.normalizedPlace,
      lat: input.lat,
      lon: input.lon,
      provider: input.provider
    },
    create: {
      key,
      query: normalizePlaceQuery(input.query),
      normalizedPlace: input.normalizedPlace,
      lat: input.lat,
      lon: input.lon,
      provider: input.provider
    }
  });
}

src/geo/nominatim.ts

type NominatimResult = {
  display_name: string;
  lat: string;
  lon: string;
};

export async function geocodeNominatim(place: string) {
  // Nominatim requires a valid User-Agent
  const url = new URL("https://nominatim.openstreetmap.org/search");
  url.searchParams.set("q", place);
  url.searchParams.set("format", "json");
  url.searchParams.set("limit", "1");
  url.searchParams.set("addressdetails", "0");

  const res = await fetch(url.toString(), {
    headers: {
      "User-Agent": "SoulCodex/1.0 (contact: you@example.com)",
      "Accept": "application/json"
    }
  });

  if (!res.ok) throw new Error(`Geocode error: ${res.status} ${await res.text()}`);

  const data = (await res.json()) as NominatimResult[];
  if (!data?.length) throw new Error("No geocode results.");

  const first = data[0];
  const lat = Number(first.lat);
  const lon = Number(first.lon);
  if (!Number.isFinite(lat) || !Number.isFinite(lon)) throw new Error("Invalid lat/lon from geocoder.");

  return {
    normalizedPlace: first.display_name,
    lat,
    lon
  };
}

src/geo/index.ts

import { getGeoCached, setGeoCached } from "./cache";
import { geocodeNominatim } from "./nominatim";

export async function resolveGeo(place: string) {
  const cached = await getGeoCached(place);
  if (cached) {
    return {
      normalizedPlace: cached.normalizedPlace,
      lat: cached.lat,
      lon: cached.lon,
      provider: cached.provider,
      cached: true
    };
  }

  const fresh = await geocodeNominatim(place);
  await setGeoCached({
    query: place,
    normalizedPlace: fresh.normalizedPlace,
    lat: fresh.lat,
    lon: fresh.lon,
    provider: "nominatim"
  });

  return {
    ...fresh,
    provider: "nominatim",
    cached: false
  };
}

4) Timezone + DST layer (offline, cached)

src/timezone/cache.ts

import { prisma } from "@/lib/prisma";

export function tzCacheKey(lat: number, lon: number, dateISO: string, time24: string) {
  const a = lat.toFixed(5);
  const o = lon.toFixed(5);
  return `tz:v1:${a}:${o}:${dateISO}:${time24}`;
}

export async function getTzCached(lat: number, lon: number, dateISO: string, time24: string) {
  const key = tzCacheKey(lat, lon, dateISO, time24);
  return prisma.timezoneCache.findUnique({ where: { key } });
}

export async function setTzCached(input: {
  lat: number;
  lon: number;
  dateISO: string;
  time24: string;
  timezone: string;
  utcOffsetMin: number;
  isDST: boolean;
}) {
  const key = tzCacheKey(input.lat, input.lon, input.dateISO, input.time24);
  return prisma.timezoneCache.upsert({
    where: { key },
    update: {
      lat: input.lat,
      lon: input.lon,
      dateISO: input.dateISO,
      time24: input.time24,
      timezone: input.timezone,
      utcOffsetMin: input.utcOffsetMin,
      isDST: input.isDST
    },
    create: {
      key,
      lat: input.lat,
      lon: input.lon,
      dateISO: input.dateISO,
      time24: input.time24,
      timezone: input.timezone,
      utcOffsetMin: input.utcOffsetMin,
      isDST: input.isDST
    }
  });
}

src/timezone/resolve.ts

import tzLookup from "tz-lookup";
import { DateTime } from "luxon";
import { getTzCached, setTzCached } from "./cache";

export async function resolveTimezone(lat: number, lon: number, dateISO: string, time24: string) {
  const cached = await getTzCached(lat, lon, dateISO, time24);
  if (cached) {
    return {
      timezone: cached.timezone,
      utcOffsetMin: cached.utcOffsetMin,
      isDST: cached.isDST,
      cached: true
    };
  }

  const timezone = tzLookup(lat, lon);

  // Build local time in that timezone
  const dt = DateTime.fromISO(`${dateISO}T${time24}`, { zone: timezone });
  if (!dt.isValid) throw new Error("Invalid date/time for timezone resolution.");

  const utcOffsetMin = dt.offset; // minutes
  const isDST = dt.isInDST;

  await setTzCached({ lat, lon, dateISO, time24, timezone, utcOffsetMin, isDST });

  return { timezone, utcOffsetMin, isDST, cached: false };
}

src/app/api/geo/lookup/route.ts

import { NextResponse } from "next/server";
import { z } from "zod";
import { resolveGeo } from "@/geo";

const Schema = z.object({ place: z.string().min(2) });

export async function POST(req: Request) {
  try {
    const body = await req.json();
    const { place } = Schema.parse(body);

    const geo = await resolveGeo(place);
    return NextResponse.json({ ok: true, geo });
  } catch (err: any) {
    return NextResponse.json({ ok: false, error: err?.message ?? "Unknown error" }, { status: 400 });
  }
}

6) Confidence labels (Verified / Partial / Unverified)

src/soulcodex/compute/confidence.ts

export type ConfidenceBadge = "verified" | "partial" | "unverified";

export function computeConfidence(input: {
  timeUnknown: boolean;
  hasGeo: boolean;
  hasTimezone: boolean;
}) {
  const { timeUnknown, hasGeo, hasTimezone } = input;

  if (!hasGeo || !hasTimezone) {
    return {
      badge: "unverified" as ConfidenceBadge,
      label: "Unverified",
      reason: "Missing geolocation/timezone lock. Rising/Houses may be inaccurate."
    };
  }

  if (timeUnknown) {
    return {
      badge: "partial" as ConfidenceBadge,
      label: "Partial",
      reason: "Birth time unknown: Rising sign and Houses are omitted."
    };
  }

  return {
    badge: "verified" as ConfidenceBadge,
    label: "Verified",
    reason: "Time + geolocation + timezone locked."
  };
}
